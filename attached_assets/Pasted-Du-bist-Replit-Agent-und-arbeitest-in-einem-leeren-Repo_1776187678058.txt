#pragma once

#include <array>
#include <optional>

class Board
{
public:
    enum class Cell
    {
        Empty,
        X,
        O
    };

    static constexpr int Size = 3;

    Board();

    void reset();

    bool isFull() const;

    Cell get(int row, int col) const;

    bool trySet(int row, int col, Cell value);

    // Returns X or O if winner exists; nullopt otherwise
    std::optional<Cell> winner() const;

private:
    std::array<Cell, Size * Size> m_cells{};

    static int index(int row, int col);
};

#include "Board.h"

Board::Board()
{
    reset();
}

void Board::reset()
{
    m_cells.fill(Cell::Empty);
}

int Board::index(int row, int col)
{
    return row * Size + col;
}

Board::Cell Board::get(int row, int col) const
{
    return m_cells.at(index(row, col));
}

bool Board::trySet(int row, int col, Cell value)
{
    const int i = index(row, col);
    if (m_cells.at(i) != Cell::Empty)
        return false;

    m_cells.at(i) = value;
    return true;
}

bool Board::isFull() const
{
    for (const auto c : m_cells)
    {
        if (c == Cell::Empty)
            return false;
    }
    return true;
}

std::optional<Board::Cell> Board::winner() const
{
    // Check rows
    for (int r = 0; r < Size; ++r)
    {
        const Cell a = get(r, 0);
        const Cell b = get(r, 1);
        const Cell c = get(r, 2);
        if (a != Cell::Empty && a == b && b == c)
            return a;
    }

    // Check columns
    for (int c = 0; c < Size; ++c)
    {
        const Cell a = get(0, c);
        const Cell b = get(1, c);
        const Cell d = get(2, c);
        if (a != Cell::Empty && a == b && b == d)
            return a;
    }

    // Diagonal top-left -> bottom-right
    {
        const Cell a = get(0, 0);
        const Cell b = get(1, 1);
        const Cell c = get(2, 2);
        if (a != Cell::Empty && a == b && b == c)
            return a;
    }

    // Diagonal top-right -> bottom-left
    {
        const Cell a = get(0, 2);
        const Cell b = get(1, 1);
        const Cell c = get(2, 0);
        if (a != Cell::Empty && a == b && b == c)
            return a;
    }

    return std::nullopt;
}

#pragma once

#include "Board.h"

class Game
{
public:

    {
        X,
        O
    };

    enum class Result
    {
        InProgress,
        XWins,
        OWins,
        Draw
    };

    Game();

    void reset();

    Player currentPlayer() const;

    Result result() const;

    const Board& board() const;

    // Returns true if move was applied
    bool makeMove(int row, int col);

private:
    Board m_board{};
    Player m_currentPlayer{Player::X};
    Result m_result{Result::InProgress};

    static Board::Cell playerToCell(Player p);

    void updateResult();
    void switchPlayer();
};

#include "Game.h"

Game::Game()
{
    reset();
}

void Game::reset()
{
    m_board.reset();
    m_currentPlayer = Player::X;
    m_result = Result::InProgress;
}

Game::Player Game::currentPlayer() const
{
    return m_currentPlayer;
}

Game::Result Game::result() const
{
    return m_result;
}

const Board& Game::board() const
{
    return m_board;
}

Board::Cell Game::playerToCell(Player p)
{
    return (p == Player::X) ? Board::Cell::X : Board::Cell::O;
}

bool Game::makeMove(int row, int col)
{
    if (m_result != Result::InProgress)
        return false;

    const Board::Cell cell = playerToCell(m_currentPlayer);

    if (!m_board.trySet(row, col, cell))
        return false;

    updateResult();

    if (m_result == Result::InProgress)
        switchPlayer();

    return true;
}

void Game::switchPlayer()
{
    m_currentPlayer = (m_currentPlayer == Player::X) ? Player::O : Player::X;
}

void Game::updateResult()
{
    if (const auto w = m_board.winner())
    {
        if (*w == Board::Cell::X)
            m_result = Result::XWins;
        else
            m_result = Result::OWins;
        return;
    }

    if (m_board.isFull())
    {
        m_result = Result::Draw;
        return;
    }

    m_result = Result::InProgress;
}

#include <SFML/Graphics.hpp>

#include <optional>
#include <string>

#include "Game.h"

namespace
{
    constexpr unsigned int WindowWidth = 600;
    constexpr unsigned int WindowHeight = 700;

    constexpr float UiTop = 0.f;
    constexpr float UiHeight = 80.f;

    constexpr float BoardLeft = 0.f;
    constexpr float BoardTop = UiHeight;
    constexpr float BoardSize = 600.f;

    constexpr float CellSize = BoardSize / 3.f;
    constexpr float LineThickness = 6.f;

    constexpr float MarkThickness = 10.f;
    constexpr float MarkMarginFactor = 0.20f; // 20% margin inside each cell

    bool pointInBoard(sf::Vector2f p)
    {
        return (p.x >= BoardLeft && p.x < BoardLeft + BoardSize &&
                p.y >= BoardTop && p.y < BoardTop + BoardSize);
    }

    bool boardCellFromPoint(sf::Vector2f p, int& outRow, int& outCol)
    {
        if (!pointInBoard(p))
            return false;

        const float localX = p.x - BoardLeft;
        const float localY = p.y - BoardTop;

        outCol = static_cast<int>(localX / CellSize);
        outRow = static_cast<int>(localY / CellSize);

        if (outRow < 0 || outRow >= 3 || outCol < 0 || outCol >= 3)
            return false;

        return true;
    }

    sf::Vector2f cellTopLeft(int row, int col)
    {
        return {BoardLeft + col * CellSize, BoardTop + row * CellSize};
    }

    sf::Vector2f cellCenter(int row, int col)
    {
        const auto tl = cellTopLeft(row, col);
        return {tl.x + CellSize * 0.5f, tl.y + CellSize * 0.5f};
    }

    std::string resultToString(const Game& game)
    {
        switch (game.result())
        {
            case Game::Result::InProgress:
            {
                const char p = (game.currentPlayer() == Game::Player::X) ? 'X' : 'O';
                return "Player: " + std::string(1, p);
            }
            case Game::Result::XWins:
                return "Winner: X";
            case Game::Result::OWins:
                return "Winner: O";
            case Game::Result::Draw:
                return "Draw!";
        }
        return "";
    }
}

int main()
{
    sf::RenderWindow window(sf::VideoMode({WindowWidth, WindowHeight}), "TicTacToe (SFML)");
    window.setFramerateLimit(60);

    // Load font (asset must be in repo AND next to binary after build copy step)
    sf::Font font;
    if (!font.openFromFile("assets/OpenSans-Regular.ttf"))
    {
        // If this fails, we cannot render text; print error and exit.
        // Keep it simple: no fallback.
        return 1;
    }

    sf::Text statusText(font, "", 28);
    statusText.setFillColor(sf::Color::White);
    statusText.setPosition({16.f, 16.f});

    sf::Text helpText(font, "Click to place. R = reset. ESC = quit.", 18);
    helpText.setFillColor(sf::Color(200, 200, 200));
    helpText.setPosition({16.f, 46.f});

    Game game;

    while (window.isOpen())
    {
        while (const std::optional event = window.pollEvent())
        {
            if (event->is<sf::Event::Closed>())
            {
                window.close();
            }

            if (const auto* key = event->getIf<sf::Event::KeyPressed>())
            {
                if (key->code == sf::Keyboard::Key::Escape)
                {
                    window.close();
                }
                else if (key->code == sf::Keyboard::Key::R)
                {
                    game.reset();
                }
            }

            if (const auto* mouse = event->getIf<sf::Event::MouseButtonPressed>())
            {
                if (mouse->button == sf::Mouse::Button::Left)
                {
                    const sf::Vector2f p{static_cast<float>(mouse->position.x),
                                         static_cast<float>(mouse->position.y)};

                    int row = -1;
                    int col = -1;
                    if (boardCellFromPoint(p, row, col))
                    {
                        game.makeMove(row, col);
                    }
                }
            }
        }

        statusText.setString(resultToString(game));

        window.clear(sf::Color(25, 25, 30));

        // Draw UI text
        window.draw(statusText);
        window.draw(helpText);

        // Draw grid lines
        for (int i = 1; i <= 2; ++i)
        {
            // Vertical lines
            {
                sf::RectangleShape vLine({LineThickness, BoardSize});
                vLine.setFillColor(sf::Color::White);
                const float x = BoardLeft + i * CellSize - LineThickness * 0.5f;
                vLine.setPosition({x, BoardTop});
                window.draw(vLine);
            }

            // Horizontal lines
            {
                sf::RectangleShape hLine({BoardSize, LineThickness});
                hLine.setFillColor(sf::Color::White);
                const float y = BoardTop + i * CellSize - LineThickness * 0.5f;
                hLine.setPosition({BoardLeft, y});
                window.draw(hLine);
            }
        }

        // Draw marks
        for (int r = 0; r < 3; ++r)
        {
            for (int c = 0; c < 3; ++c)
            {
                const auto cell = game.board().get(r, c);
                if (cell == Board::Cell::Empty)
                    continue;

                const sf::Vector2f center = cellCenter(r, c);
                const float margin = CellSize * MarkMarginFactor;
                const float length = CellSize - 2.f * margin;

                if (cell == Board::Cell::X)
                {
                    sf::RectangleShape d1({length, MarkThickness});
                    d1.setFillColor(sf::Color(220, 70, 70));
                    d1.setOrigin({length * 0.5f, MarkThickness * 0.5f});
                    d1.setPosition(center);
                    d1.setRotation(sf::degrees(45.f));

                    sf::RectangleShape d2({length, MarkThickness});
                    d2.setFillColor(sf::Color(220, 70, 70));
                    d2.setOrigin({length * 0.5f, MarkThickness * 0.5f});
                    d2.setPosition(center);
                    d2.setRotation(sf::degrees(-45.f));

                    window.draw(d1);
                    window.draw(d2);
                }
                else if (cell == Board::Cell::O)
                {
                    const float radius = (length * 0.5f);
                    sf::CircleShape circle(radius);
                    circle.setFillColor(sf::Color::Transparent);
                    circle.setOutlineThickness(MarkThickness);
                    circle.setOutlineColor(sf::Color(70, 180, 230));
                    circle.setOrigin({radius, radius});
                    circle.setPosition(center);

                    window.draw(circle);
                }
            }
        }

        window.display();
    }

    return 0;
}

# TicTacToe (C++17 + SFML)

## Kurzbeschreibung
Ein kleines TicTacToe-Spiel (3x3) geschrieben in C++17. Rendering, Input und Fenster laufen über SFML.

## Features
- 2-Spieler lokal (X startet)
- Maus-Klick setzt X/O
